189/*
190 * Auxiliary function to add an outgoing edge to a vertex (this),
191 * with a given destination vertex (d) and edge info (info).
192 */
193template <class T, class W>
194Edge<T, W> *Vertex<T, W>::addEdge(Vertex<T, W> *destination, W info) {
195 if (destination == nullptr) return nullptr;
196 Edge<T, W> *newEdge = new Edge<T, W>(this, destination, info);
197 this->_outgoing.push_back(newEdge);
198 destination->_incoming.push_back(newEdge);
199 return newEdge;
200}

202/*
203 * Auxiliary function that removes an edge pointer from it's outgoing
204 * list of edges. Returns true if removed, false otherwise.
205 */
206template <class T, class W>
207bool Vertex<T, W>::removeOutgoingEdge(Edge<T, W> *edge) {
208 auto it = this->_outgoing.begin();
209 while (it != this->_outgoing.end()) {
210 if ((*it) == edge) {
211 this->_outgoing.erase(it);
212 return true;
213 }
214 it++;
215 }
216 return false;
217}

300/*
301 * Removes a vertex (vertex) from a graph (this), and
302 * all outgoing and incoming edges.
303 * Returns true if successful, and false if such vertex does not exist.
304 */
305template <class T, class W>
306bool Graph<T, W>::removeVertex(Vertex<T, W> *vertex) {
307 for (auto it = this->_vertexSet.begin(); it != this->_vertexSet.end(); it++) {
308 if ((*it) == vertex) {
309 delete (*it); // Delete Vertex
310 it = this->_vertexSet.erase(it); // Remove pointer from list
311 return true;
312 }
313 }
314 return false;
315}

342template <class T, class W>
343Edge<T, W> *Graph<T, W>::addBidirectionalEdge(Vertex<T, W> *vertexA, Vertex<T, W> *vertexB, const W &info) {
344 if (vertexA == nullptr || vertexB == nullptr) return nullptr;
345 Edge<T, W> *edgeA = vertexA->addEdge(vertexB, info);
346 Edge<T, W> *edgeB = vertexB->addEdge(vertexA, info);
347 edgeA->setReverse(edgeB);
348 edgeB->setReverse(edgeA);

352/*
353 * Removes an edge from a graph (this).
354 * The edge is identified by the source (sourc) and destination (dest) contents.
355 * Returns true if successful, and false if such edge does not exist.
356 */
357template <class T, class W>
358bool Graph<T, W>::removeEdge(Edge<T, W> *edge) {
359 if (edge == nullptr) return false;
360
361 // Check if Edge belongs to the graph
362 bool foundSource = std::find(_vertexSet.begin(), _vertexSet.end(), edge->getOrigin()) != _vertexSet.end();
363 bool foundDestination = std::find(_vertexSet.begin(), _vertexSet.end(), edge->getDestination()) != _vertexSet.end();
364
365 // Delete Edge
366 if (foundSource and foundDestination) {
367 delete edge;
368 return true;
369 }
370 return false;
371}
